<script>
(function() {
  var CONSENT_KEY = 'arcadedb_cookie_consent';

  function hasConsented() {
    if (localStorage.getItem(CONSENT_KEY)) return true;
    var cookies = document.cookie.split(';');
    for (var i = 0; i < cookies.length; i++) {
      if (cookies[i].trim().startsWith(CONSENT_KEY + '=')) return true;
    }
    return false;
  }

  function storeConsent() {
    localStorage.setItem(CONSENT_KEY, '1');
    var expires = new Date(Date.now() + 365 * 24 * 60 * 60 * 1000).toUTCString();
    document.cookie = CONSENT_KEY + '=1; expires=' + expires + '; path=/; domain=.arcadedb.com; SameSite=Lax';
  }

  if (!hasConsented()) {
    var banner = document.getElementById('arcadedb-cookie-banner');
    banner.style.display = 'flex';
    document.getElementById('arcadedb-cookie-ok').addEventListener('click', function() {
      storeConsent();
      banner.style.display = 'none';
    });
  }
})();
</script>
