  var hash = window.location.hash;

  if (hash) {
    hash = hash.substr(1); // Remove "#" hash character
    $("a[href$='" + hash + "']").each(function() {
      var title = this.text;

      let isSection = /^\d/.test(title);
      if (isSection) {
        $('title').text(title + " - ArcadeDB Manual");
      }
    });
  }
});
